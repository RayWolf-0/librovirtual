CREATE TABLE cursos (
    id_curso NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nivel VARCHAR2(50) NOT NULL,
    letra CHAR(1) NOT NULL,
    anio NUMBER(4) NOT NULL
);

CREATE TABLE asignaturas (
    id_asignatura NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre_asignatura VARCHAR2(100) NOT NULL
);

CREATE TABLE estudiantes (
    id_estudiante NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    rut VARCHAR2(12) UNIQUE NOT NULL,
    nombre VARCHAR2(100) NOT NULL,
    apellido VARCHAR2(100) NOT NULL,
    id_curso NUMBER,
    CONSTRAINT fk_estudiante_curso FOREIGN KEY (id_curso) REFERENCES cursos(id_curso)
);

CREATE TABLE apoderados (
    id_apoderado NUMBER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
    id_usuario NUMBER NOT NULL, -- Relación con la tabla general de usuarios
    telefono_contacto VARCHAR2(20),
    parentesco_titular VARCHAR2(20), -- 'Padre', 'Madre', 'Tutor'
    CONSTRAINT fk_apoderado_usuario FOREIGN KEY (id_usuario) REFERENCES usuarios_comunidad(id_usuario)
);

CREATE TABLE evaluaciones (
    id_evaluacion NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre_evaluacion VARCHAR2(100),
    fecha DATE NOT NULL,
    ponderacion NUMBER(3,2), -- Ejemplo: 0.25 para un 25%
    id_asignatura NUMBER,
    CONSTRAINT fk_eval_asignatura FOREIGN KEY (id_asignatura) REFERENCES asignaturas(id_asignatura)
);

CREATE TABLE calificaciones (
    id_calificacion NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    valor_nota NUMBER(3,1), -- Ejemplo: 7.0
    id_estudiante NUMBER,
    id_evaluacion NUMBER,
    CONSTRAINT fk_calif_estudiante FOREIGN KEY (id_estudiante) REFERENCES estudiantes(id_estudiante),
    CONSTRAINT fk_calif_evaluacion FOREIGN KEY (id_evaluacion) REFERENCES evaluaciones(id_evaluacion)
);

-- ==========================================================
-- BLOQUE 2: REGISTRO DE ASISTENCIA Y ANOTACIONES
-- ==========================================================

CREATE TABLE asistencia (
    id_asistencia NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    fecha DATE DEFAULT SYSDATE,
    estado CHAR(1), -- 'P' Presente, 'A' Ausente
    id_estudiante NUMBER NOT NULL,
    id_asignatura NUMBER NOT NULL,
    CONSTRAINT fk_asist_estudiante FOREIGN KEY (id_estudiante) REFERENCES estudiantes(id_estudiante),
    CONSTRAINT fk_asist_asignatura FOREIGN KEY (id_asignatura) REFERENCES asignaturas(id_asignatura)
);

CREATE TABLE anotaciones (
    id_anotacion NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    tipo VARCHAR2(20), -- 'Positiva', 'Negativa'
    descripcion CLOB,
    fecha DATE DEFAULT SYSDATE,
    id_estudiante NUMBER NOT NULL,
    CONSTRAINT fk_anot_estudiante FOREIGN KEY (id_estudiante) REFERENCES estudiantes(id_estudiante)
);

-- ==========================================================
-- BLOQUE 3: PORTAL DE COMUNICACIONES (MENSAJERÍA)
-- ==========================================================

CREATE TABLE usuarios_comunidad (
    id_usuario NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre_completo VARCHAR2(200),
    correo VARCHAR2(150) UNIQUE,
    rol VARCHAR2(50) -- 'Docente', 'Apoderado', 'Estudiante'
);

CREATE TABLE mensajes (
    id_mensaje NUMBER GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    asunto VARCHAR2(200),
    contenido CLOB,
    fecha_envio TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    id_emisor NUMBER,
    id_receptor NUMBER,
    CONSTRAINT fk_msj_emisor FOREIGN KEY (id_emisor) REFERENCES usuarios_comunidad(id_usuario),
    CONSTRAINT fk_msj_receptor FOREIGN KEY (id_receptor) REFERENCES usuarios_comunidad(id_usuario)
);
